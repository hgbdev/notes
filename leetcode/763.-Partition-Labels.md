# 763. Partition Labels

function partitionLabels(s: string): number[] {
    const results: number[] = [];
    const last: any = {};
    let anchor: number = 0;
    let j: number = 0;

    for(let i = 0; i < s.length; i++) {
        last[s[i]] = i;
    }

    for(let i = 0; i < s.length; i++) {
        j = Math.max(j, last[s[i]]);
        if(i === j) {
            results.push(i - anchor + 1);
            anchor = i + 1;
        }
    }

    return results;
};
