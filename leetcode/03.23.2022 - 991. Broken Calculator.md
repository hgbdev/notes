//Typescript
function brokenCalc(startValue: number, target: number): number {
    let result: number = 0;
    let _tTarget: number = target;

    while(_tTarget > startValue) {
        result++;
        if(_tTarget % 2 === 0) {
            _tTarget = _tTarget / 2;
        } else {
            _tTarget++;
        }
    }

    return result + startValue - _tTarget;
};
